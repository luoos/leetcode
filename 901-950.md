## [912. Sort an Array - Medium](https://leetcode.com/problems/sort-an-array/)

class Solution:
  def sortArray(self, nums: List[int]) -> List[int]:
    # self.mergeSort(nums)
    self.quickSort(nums, 0, len(nums)-1)
    return nums

  def mergeSort(self, nums):
    if len(nums) <= 1:
      return
    mid = len(nums) // 2
    L = nums[:mid]
    R = nums[mid:]
    self.mergeSort(L)
    self.mergeSort(R)
    i = left = right = 0
    while left < len(L) and right < len(R):
      if L[left] < R[right]:
        nums[i] = L[left]
        left += 1
      else:
        nums[i] = R[right]
        right += 1
      i += 1
    if left < len(L):
      nums[i:] = L[left:]
    if right < len(R):
      nums[i:] = R[right:]
