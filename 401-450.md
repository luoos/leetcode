## [415. Add Strings - Easy](https://leetcode.com/problems/add-strings/)

```python
class Solution:
  def addStrings(self, num1: str, num2: str) -> str:
    slots = []
    i, j = len(num1) - 1, len(num2) - 1
    carry = 0
    while i >= 0 or j >= 0 or carry:
      d1 = d2 = 0
      if i >= 0:
        d1 = int(num1[i])
        i -= 1
      if j >= 0:
        d2 = int(num2[j])
        j -= 1

      carry, v = divmod(carry+d1+d2, 10)
      slots.append(str(v))
    return ''.join(slots[::-1])
```

## [444. Sequence Reconstruction - Medium](https://leetcode.com/problems/sequence-reconstruction/)

```python
# topology sort
from collections import defaultdict

class Solution:
  def sequenceReconstruction(self, org: List[int], seqs: List[List[int]]) -> bool:
    nexts = defaultdict(set)
    in_degree = defaultdict(int)
    all_nums = set()
    for s in seqs:
      all_nums |= set(s)
      if s:
        in_degree[s[0]] += 0
      for before, after in zip(s, s[1:]):
        if after not in nexts[before]:
          nexts[before].add(after)
          in_degree[after] += 1
    cur = [k for k, v in in_degree.items() if v == 0]
    res = []
    while len(cur) == 1:
      n = cur.pop()
      res.append(n)
      for nex in nexts[n]:
        in_degree[nex] -= 1
        if in_degree[nex] == 0:
          cur.append(nex)
    if len(cur) > 1:
      return False
    return res == org and len(res) == len(all_nums)
```
